\section{$I_{n1}$}

$I_{n1} = \int_{\Sr} [\hat{f}_{h_1, \dots, h_r}(\mathbf{x}) - \mathsf{E} (\hat{f}_{h_1, \dots, h_r}(\mathbf{x}))]^2 \mathrm{d} \mathbf{x}$.

\begin{align}
    I_{n1} &= \int_{\Sr} \Bigg[ \frac{1}{n} \sum_{i=1}^{n} \prod_{j=1}^{r} L_{\mathbf{h}} (x,X_j^{(i)}) - \mathsf{E} \Big( \frac{1}{n} \sum_{i=1}^{n} \prod_{j=1}^{r} L_{\mathbf{h}} (x,X_j^{(i)}) \Big) \Bigg]^2 \mathrm{d} \mathbf{x} \\
    &= \frac{1}{n^2} \int_{\Sr} \Bigg[ \sum_{i=1}^{n} \prod_{j=1}^{r} L_{\mathbf{h}} (x,X_j^{(i)}) - \mathsf{E} \Big( \sum_{i=1}^{n} \prod_{j=1}^{r} L_{\mathbf{h}} (x,X_j^{(i)}) \Big) \Bigg]^2 \mathrm{d} \mathbf{x} \\
    &= \frac{1}{n^2} \int_{\Sr} \Bigg[ \sum_{i=1}^{n} \prod_{j=1}^{r} L_{\mathbf{h}} (x,X_j^{(i)}) - \sum_{i=1}^{n} \mathsf{E} \Big( \prod_{j=1}^{r} L_{\mathbf{h}} (x,X_j^{(i)}) \Big) \Bigg]^2 \mathrm{d} \mathbf{x} \\
    &= \frac{1}{n^2} \int_{\Sr} \Bigg[ \sum_{i=1}^{n} \Bigg\{ \prod_{j=1}^{r} L_{\mathbf{h}} (x,X_j^{(i)}) - \mathsf{E} \Big( \prod_{j=1}^{r} L_{\mathbf{h}} (x,X_j^{(i)}) \Big) \Bigg\} \Bigg]^2 \mathrm{d} \mathbf{x} \\
    &= \frac{1}{n^2} \int_{\Sr} \Bigg[ \sum_{i=1}^{n} \Bigg\{ \prod_{j=1}^{r} L_{\mathbf{h}} (x,X_j^{(i)}) - \prod_{j=1}^{r} \mathsf{E} \Big( L_{\mathbf{h}} (x,X_j^{(i)}) \Big) \Bigg\} \Bigg]^2 \mathrm{d} \mathbf{x},
\end{align}
where the last equality comes from the null hypothesis of independence between the elements of the polysphere. Therefore, $I_{n1} = I_{n1}^{(1)} + I_{n1}^{(2)}$, where
\begin{align}
    I_{n1}^{(1)} &:= \frac{1}{n^2} \int_{\Sr} \Bigg[ \sum_{i=1}^{n} \Bigg\{ \prod_{j=1}^{r} L_{\mathbf{h}} (x,X_j^{(i)}) - \prod_{j=1}^{r} \mathsf{E} \Big( L_{\mathbf{h}} (x,X_j^{(i)}) \Big) \Bigg\}^2 \Bigg] \mathrm{d} \mathbf{x}; \\
    I_{n1}^{(2)} &:= \frac{2}{n^2} \int_{\Sr} \Bigg[ \sum_{1 \le i<k \le n} \Bigg\{ \prod_{j=1}^{r} L_{\mathbf{h}} (x,X_j^{(i)}) - \prod_{j=1}^{r} \mathsf{E} \Big( L_{\mathbf{h}} (x,X_j^{(i)}) \Big) \Bigg\} \\
    &\times \Bigg\{ \prod_{j=1}^{r} L_{\mathbf{h}} (x,X_j^{(k)}) - \prod_{j=1}^{r} \mathsf{E} \Big( L_{\mathbf{h}} (x,X_j^{(k)}) \Big) \Bigg\} \Bigg] \mathrm{d} \mathbf{x}.
\end{align}
The asymptotic distribution of $I_{n1} = I_{n1}^{(1)} + I_{n1}^{(2)}$ can be described through Lemma 1 in \cite{Garc_a_Portugu_es_2015}. In this case,
\begin{align}
    B_n^{-1/2} I_{n1} \xrightarrow{d} \mathcal{N}(0, 1),
\end{align}
where $B_n = n \mathsf{E} (\varphi_n^2(X_1)) + \frac{1}{2} n^2 \mathsf{E}(H_n^2(X_1,X_2))$ and $\varphi_n(X_1) = \frac{1}{n^2} \int_{\Sr} \Bigg[ \Bigg\{ \prod_{j=1}^{r} L_{\mathbf{h}} (x,X_j^{(1)}) - \prod_{j=1}^{r} \mathsf{E} \Big( L_{\mathbf{h}} (x,X_j^{(1)}) \Big) \Bigg\}^2 \Bigg] \mathrm{d} \mathbf{x}$.

\subsubsection{Second term}

\begin{align*}

Note that, by Cuachy-Schwarz inequality,
\begin{align*}
    \mathsf{E} (\bar{L}_{\bh} (\bt,\bX_1) \bar{L}_{\bh} (\bs,\bX_1)) & \le \sqrt{\mathsf{E}^2 (\bar{L}_{\bh} (\bt,\bX_1)) \mathsf{E}^2 (\bar{L}_{\bh} (\bs,\bX_1)))} \\
    &= \sqrt{\mathsf{Var} ({L}_{\bh} (\bt,\bX_1)) \mathsf{Var} ({L}_{\bh} (\bs,\bX_1)))}.
\end{align*}

Therefore,
\begin{align*}
    G_n(x,y) & \le \frac{4}{n^4} \int_{\Sr} \int_{\Sr} \bar{L}_{\bh} (\bt,\bx) \bar{L}_{\bh} (\bs,\by) \sqrt{\mathsf{Var} ({L}_{\bh} (\bt,\bX_1)) \mathsf{Var} ({L}_{\bh} (\bs,\bX_1)))} \,\rd \bt \,\rd \bs. \\

\end{align*}

Therefore,
\begin{align*}
    \mathsf{E} (G_n^2(X_1,X_2)) & \le \frac{16}{n^{16}} \int_{(\Sr)^4} \mathsf{E} \lrb{ \bar{L}_{\bh} (\bt_1,\bX_1) \bar{L}_{\bh} (\bt_2,\bX_2) \bar{L}_{\bh} (\bt_3,\bX_1) \bar{L}_{\bh} (\bt_4,\bX_2)} \sqrt{\prod_{i=1}^{4} \mathsf{Var} ({L}_{\bh} (\bt_i,\bX_1)) } \prod_{i=1}^{4} \,\rd \bt_i \\

\end{align*}

\begin{align*}
    H_n^4(\bX_1, \bX_2) &= \frac{16}{n^{8}} \lrp{\int_{\Sr} \bar{L}_{\bh} (\bx,\bX_1) \bar{L}_{\bh} (\bx,\bX_2) \,\rd \bx}^4\\
    &=\frac{16}{n^{8}}\int_{(\Sr)^4} \Bigg( \prod_{l=1}^{4} \bar{L}_{\bh} (\bx_l,\bX_1) \Bigg) \Bigg( \prod_{l=1}^{4} \bar{L}_{\bh} (\bx_l,\bX_2) \Bigg) \prod_{l=1}^{4} \,\rd 
    \bx_l.
\end{align*}

\begin{align*}
    \mathsf{E} (H_n^4(\bX_1, \bX_2)) &= \frac{16}{n^{8}} \int_{(\Sr)^4} \mathsf{E}^2 \Bigg( \prod_{l=1}^{4} \bar{L}_{\bh} (\bx_l,\bX_1) \Bigg) \prod_{l=1}^{4} \,\rd 
    \bx_l \\
    & \le \frac{16}{n^{8}} \int_{(\Sr)^4} \mathsf{E} \Bigg( \prod_{l=1}^{4} \bar{L}_{\bh}^2 (\bx_l,\bX_1) \Bigg) \prod_{l=1}^{4} \,\rd \bx_l \\
    &= \frac{16}{n^{8}} \mathsf{E} \Bigg[ \int_{(\Sr)^4} \Bigg( \prod_{l=1}^{4} \bar{L}_{\bh}^2 (\bx_l,\bX_1) \Bigg) \prod_{l=1}^{4} \,\rd \bx_l \Bigg] \\
    &= \frac{16}{n^{8}} \mathsf{E} \Bigg[ \Bigg( \int_{\Sr} \bar{L}_{\bh}^2 (\bx,\bX_1) \,\rd \bx \Bigg)^4 \Bigg] \\
    &= \frac{16}{n^{8}} \mathsf{E} (Z_{n1}^4).
\end{align*}

Applying the $c_p$ inequality two times, we have that
\begin{align*}
    Z_{n1}^4 & \le \Bigg( \int_{\Sr} 2[L_{\bh}^2 (\bx,\bX_1) + \mathsf{E}^2 (L_{\bh} (\bx,\bX_1))] \rd \bx \Bigg)^4 \\
    &= 16 \Bigg( \int_{\Sr} L_{\bh}^2 (\bx,\bX_1) \rd \bx + \int_{\Sr} \mathsf{E}^2 (L_{\bh} (\bx,\bX_1)) \rd \bx \Bigg)^4 \\
    & \le 128 \Bigg( \Bigg( \int_{\Sr} L_{\bh}^2 (\bx,\bX_1) \rd \bx \Bigg)^4 + \Bigg( \int_{\Sr} \mathsf{E}^2 (L_{\bh} (\bx,\bX_1)) \rd \bx \Bigg)^4 \Bigg).
\end{align*}

Now applying Lemma \ref{lemma-equivLh}, 
\begin{align*}
    \int_{\Sr} L_{\bh}^2 (\bx,\bX_1) \rd \bx \asymp \frac{\lambda_{\bd}(L^2)}{\lambda_{\bd}(L)^2} \rho(\bh)^{-1}.
\end{align*}
\begin{align*}
    \int_{\Sr} \mathsf{E}^2 (L_{\bh} (\bx,\bX_i)) &= \int_{\Sr} \Bigg( \int_{\Sr} L_{\bh} (\bx,\bX_1) f(\bx) \rd \bx \Bigg)^2 \rd \bx \\
    & \asymp \int_{\Sr} \Bigg( \frac{\lambda_{\bd}(L)}{\lambda_{\bd}(L)} f(\bx) \Bigg)^2 \rd \bx \\
    &= R(f).
\end{align*}

Therefore,
\begin{align*}
    \mathsf{E} (H_n^4(\bX_1, \bX_k)) &= \frac{16}{n^{8}} O(\rho(\bh)^{-4} + R(f)^4) \\
    &= O(n^{-8} \rho(\bh)^{-4}).
\end{align*}

\textit{Asymptotic behavior of $\mathsf{E} (G_n^2(\bX_1,\bX_2))$.} Consider the following calculations.

\begin{align*}
    H_n(\bx,\bX_1) &= \frac{2}{n^2} \int_{\Sr} \bar{L}_{\bh} (\bt,\bx) \bar{L}_{\bh} (\bt,\bX_1) \,\rd \bt.
\end{align*}

\begin{align*}
    G_n(\bx,\by) &= \mathsf{E} (H_n(\bX_1,\bx)H_n(\bX_1,\by))\\
    &=\int_{\Sr} \lrb{\frac{2}{n^2} \int_{\Sr} \bar{L}_{\bh} (\bt,\bx) \bar{L}_{\bh} (\bt,\bx_1) \,\rd \bt} \lrb{\frac{2}{n^2} \int_{\Sr} \bar{L}_{\bh} (\bs,\by) \bar{L}_{\bh} (\bs,\bx_1) \,\rd \bs} f(\bx_1) \,\rd \bx_1 \\
    &= \frac{4}{n^4} \int_{\Sr} \int_{\Sr} \int_{\Sr} \bar{L}_{\bh} (\bt,\bx) \bar{L}_{\bh} (\bt,\bx_1) \bar{L}_{\bh} (\bs,\by) \bar{L}_{\bh} (\bs,\bx_1) f(\bx_1) \,\rd \bx_1 \,\rd \bt \,\rd \bs \\
    &= \frac{4}{n^4} \int_{\Sr} \int_{\Sr} \bar{L}_{\bh} (\bt,\bx) \bar{L}_{\bh} (\bs,\by) \mathsf{E} (\bar{L}_{\bh} (\bt,\bX_1) \bar{L}_{\bh} (\bs,\bX_1)) \,\rd \bt \,\rd \bs.
\end{align*}

Note that, by Cauchy--Schwarz inequality,
\begin{align*}
    \mathsf{E} (\bar{L}_{\bh} (\bt,\bX_1) \bar{L}_{\bh} (\bs,\bX_1)) & \le \sqrt{\mathsf{E}^2 (\bar{L}_{\bh} (\bt,\bX_1)) \mathsf{E}^2 (\bar{L}_{\bh} (\bs,\bX_1))} \\
    &= \sqrt{\mathsf{Var} ({L}_{\bh} (\bt,\bX_1)) \mathsf{Var} ({L}_{\bh} (\bs,\bX_1))}.
\end{align*}

Therefore,
\begin{align*}
    G_n(\bx,\by) & \le \frac{4}{n^4} \int_{(\Sr)^2}  \bar{L}_{\bh} (\bt,\bx) \bar{L}_{\bh} (\bs,\by) \sqrt{\mathsf{Var} ({L}_{\bh} (\bt,\bX_1)) \mathsf{Var} ({L}_{\bh} (\bs,\bX_1))} \,\rd \bt \,\rd \bs. \\
    G_n^2(\bx,\by) & \le \frac{16}{n^{8}} \int_{(\Sr)^4} \bar{L}_{\bh} (\bt_1,\bx) \bar{L}_{\bh} (\bt_2,\by) \bar{L}_{\bh} (\bt_3,\bx) \bar{L}_{\bh} (\bt_4,\by) \sqrt{\prod_{i=1}^{4} \mathsf{Var} ({L}_{\bh} (\bt_i,\bX_1)) } \prod_{i=1}^{4} \,\rd \bt_i.
\end{align*}

Therefore,
\begin{align*}
    \mathsf{E} (G_n^2(\bX_1,\bX_2)) & \le \frac{16}{n^{8}} \int_{(\Sr)^4} \mathsf{E} \lrb{ \bar{L}_{\bh} (\bt_1,\bX_1) \bar{L}_{\bh} (\bt_2,\bX_2) \bar{L}_{\bh} (\bt_3,\bX_1) \bar{L}_{\bh} (\bt_4,\bX_2)} \sqrt{\prod_{i=1}^{4} \mathsf{Var} ({L}_{\bh} (\bt_i,\bX_1)) } \prod_{i=1}^{4} \,\rd \bt_i \\
    &= \frac{16}{n^{8}} \int_{(\Sr)^4} \mathsf{E} \lrb{ \bar{L}_{\bh} (\bt_1,\bX_1) \bar{L}_{\bh} (\bt_3,\bX_1) } \mathsf{E} \lrb{ \bar{L}_{\bh} (\bt_2,\bX_2) \bar{L}_{\bh} (\bt_4,\bX_2)} \sqrt{\prod_{i=1}^{4} \mathsf{Var} ({L}_{\bh} (\bt_i,\bX_1)) } \prod_{i=1}^{4} \,\rd \bt_i \\
    &= \frac{16}{n^{8}} \int_{(\Sr)^4} \mathsf{E} \lrb{ \bar{L}_{\bh} (\bt_1,\bX_1) \bar{L}_{\bh} (\bt_3,\bX_1) } \mathsf{E} \lrb{ \bar{L}_{\bh} (\bt_2,\bX_1) \bar{L}_{\bh} (\bt_4,\bX_1)} \sqrt{\prod_{i=1}^{4} \mathsf{Var} ({L}_{\bh} (\bt_i,\bX_1)) } \prod_{i=1}^{4} \,\rd \bt_i. \\
\end{align*}

Applying Cauchy--Schwartz inequality again, we obtain
\begin{align*}
    \mathsf{E} (G_n^2(\bX_1,\bX_2)) \le &\; \frac{16}{n^{8}} \int_{(\Sr)^4}  \sqrt{\mathsf{Var} ({L}_{\bh} (\bt_1,\bX_1)) \mathsf{Var} ({L}_{\bh} (\bt_3,\bX_1))} \sqrt{\mathsf{Var} ({L}_{\bh} (\bt_2,\bX_1)) \mathsf{Var} ({L}_{\bh} (\bt_4,\bX_1))} \\
    & \times \sqrt{\prod_{i=1}^{4} \mathsf{Var} ({L}_{\bh} (\bt_i,\bX_1)) } \prod_{i=1}^{4} \,\rd \bt_i \\
    =&\; \frac{16}{n^{8}} \int_{(\Sr)^4} \sqrt{\prod_{i=1}^{4} \mathsf{Var} ({L}_{\bh} (\bt_i,\bX_1)) } \sqrt{\prod_{i=1}^{4} \mathsf{Var} ({L}_{\bh} (\bt_i,\bX_1)) } \prod_{i=1}^{4} \,\rd \bt_i \\
    =&\; \frac{16}{n^{8}} \int_{(\Sr)^4} \prod_{i=1}^{4} \mathsf{Var} ({L}_{\bh} (\bt_i,\bX_1)) \prod_{i=1}^{4} \,\rd \bt_i \\
    =&\; \frac{16}{n^{8}} \int_{(\Sr)^4} \prod_{i=1}^{4} n \mathsf{Var} (\hat{f}(\bt_i;\bh)) \prod_{i=1}^{4} \,\rd \bt_i \\
    =&\; \frac{16}{n^{8}} \int_{(\Sr)^4} \prod_{i=1}^{4} n \lrb{\frac{v_{\bd}(L)}{n \rho(\bh)} f(\bt_i)(1 + o(1))} \prod_{i=1}^{4} \,\rd \bt_i \\
    =&\; \frac{16}{n^{8}} \int_{(\Sr)^4} \prod_{i=1}^{4} \frac{v_{\bd}(L)}{ \rho(\bh)} f(\bt_i)(1 + o(1)) \prod_{i=1}^{4} \,\rd \bt_i \\
    =&\; \frac{16}{n^{8}} \prod_{i=1}^{4} \int_{(\Sr)^4} \frac{v_{\bd}(L)}{ \rho(\bh)} f(\bt_i)(1 + o(1)) \,\rd \bt_i \\
    =&\; \frac{16}{n^{8}} \prod_{i=1}^{4} \frac{v_{\bd}(L)}{ \rho(\bh)} (1 + o(1)) \\
    =&\; \frac{16}{n^{8}} \frac{v_{\bd}^4(L)}{ \rho^4(\bh)} (1 + o(1)).
\end{align*}

As a consequence,
\begin{align*}

\end{align*}

    \begin{align*}
        f(\bx) &= \int_{\Sr} c^\mathrm{vMF}_{d}(\kappa_1) e^{\kappa_1\bmu^{\top}\bmu_1} \prod_{j=1}^{r} c^\mathrm{vMF}_{d}(\kappa) e^{\kappa\bx^{(j) \top}\bmu} \,\rd \bmu \\
        &= c^\mathrm{vMF}_{d}(\kappa_1) c^\mathrm{vMF}_{d}(\kappa)^r \int_{\Sr} e^{\kappa_1\bmu^{\top}\bmu_1 + \sum_{j=1}^{r} \kappa\bx^{(j) \top}\bmu} \,\rd \bmu\\
        &= c^\mathrm{vMF}_{d}(\kappa_1) c^\mathrm{vMF}_{d}(\kappa)^r \int_{\Sr} e^{\|\kappa_1\bmu_1 + \sum_{j=1}^{r} \kappa\bx^{(j)}\|\bmu^{\top}(\kappa_1\bmu_1 + \sum_{j=1}^{r} \kappa\bx^{(j)})/\|\kappa_1\bmu_1 + \sum_{j=1}^{r} \kappa\bx^{(j)}\|} \,\rd \bmu\\
        &= \frac{c^\mathrm{vMF}_{d}(\kappa_1) c^\mathrm{vMF}_{d}(\kappa)^r}{ c^\mathrm{vMF}_{d}(\|\kappa_1\bmu_1 + \sum_{j=1}^{r} \kappa\bx^{(j)}\|)^r}.
\end{align*}
%{\color{red} Relation with \cite{Johnson1978,Wehrly1980}'s copula model for $d_1=d_2=1$ and $r=2$?}
It is worth noting that, for the case with $r=2, d_1 = d_2 = 1, \kappa_1 = 0$ (two random vectors supported in the circle), the joint distribution of the angles belongs to the  Johnson-Wehrly family of distribution - see \cite{Johnson1978,Wehrly1980}. This result is available in Proposition \ref{prop-js-family}.
    For convenience, we will reparametrize the model as $\delta \defin 1/\kappa_1$, because, in this way, for fixed parameters $\bmu_1$ and $\kappa$, the parameter $\delta$ will quantify the amount of deviation from independence. By convention, we will define that, when $\delta = 0$, then $\bmu = \bmu_1$ almost surely (degenerate random vector); therefore,
    \begin{align*}
        f(\bx) &= c^\mathrm{vMF}_{d}(1/\delta) c^\mathrm{vMF}_{d}(\kappa)^r \int_{\Sr} e^{(1/\delta)\bmu^{\top}\bmu_1 + \sum_{j=1}^{r} \kappa\bx^{(j) \top}\bmu} \,\rd \bmu,
    \end{align*}
    and, for $\delta = 0$,
    \begin{align*}
        f(\bx) &= \prod_{j=1}^{r} c^\mathrm{vMF}_{d}(\kappa) e^{\kappa\bx^{(j) \top}\bmu_1} \\
        &= c^\mathrm{vMF}_{d}(\kappa)^r e^{\sum_{j=1}^{r} \kappa\bx^{(j) \top}\bmu_1},
    \end{align*}
    which is the independence case.

    The following scenarios in the sphere are considered.
    \begin{enumerate}
        \item $r=3, d=2, \bmu_1 = (\sqrt{1/3})_{1 \times 9}, \kappa = 10, \delta = 0$ (independence case);
        \item $r=3, d=2, \bmu_1 = (\sqrt{1/3})_{1 \times 9}, \kappa = 10, \delta = 0.1$;
        \item $r=3, d=2, \bmu_1 = (\sqrt{1/3})_{1 \times 9}, \kappa = 10, \delta = 0.5$;
        \item $r=3, d=2, \bmu_1 = (\sqrt{1/3})_{1 \times 9}, \kappa = 10, \delta = 1$;
        \item $r=3, d=2, \bmu_1 = (\sqrt{1/3})_{1 \times 9}, \kappa = 10, \delta = 10$.
    \end{enumerate}
     The following scenarios in the circle are considered
    \begin{enumerate}
        \item $r=2, d=1, \bmu_1 = (\sqrt{1/2})_{1 \times 4}, \kappa_1 = 0, \kappa = 0$ (independence case);
        \item $r=2, d=1, \bmu_1 = (\sqrt{1/2})_{1 \times 4}, \kappa_1 = 0, \kappa = 1$;
        \item $r=2, d=1, \bmu_1 = (\sqrt{1/2})_{1 \times 4}, \kappa_1 = 0, \kappa = 2$;
        \item $r=2, d=1, \bmu_1 = (\sqrt{1/2})_{1 \times 4}, \kappa_1 = 0, \kappa = 5$;
        \item $r=2, d=1, \bmu_1 = (\sqrt{1/2})_{1 \times 4}, \kappa_1 = 0, \kappa = 10$.
    \end{enumerate}
    \item \textbf{Markovian process:} we simulate $\bX^{(1)} \sim \mathrm{vMF}_d (\bmu_1, \kappa_1)$, and, then, we simulate $\bX^{(j)} | (\bX^{(1)} = \bx^{(1)} , \dots, \bX^{(j-1)} = \bx^{(j-1)}) \sim \mathrm{vMF}_d (\bx^{(j-1)}, \delta)$ for $j = 2, \dots, r$. This produces the density
    \begin{align*}
        f(\bx) &= c^\mathrm{vMF}_{d}(\kappa_1) e^{\kappa_1\bx^{(1) \top}\bmu_1} \prod_{j=2}^{r} c^\mathrm{vMF}_{d}(\delta) e^{\delta\bx^{(j) \top}\bx^{(j-1)}} \\
        &= c^\mathrm{vMF}_{d}(\kappa_1) c^\mathrm{vMF}_{d}(\kappa)^{r-1} e^{\kappa_1\bx^{(1) \top}\bmu_1 + \sum_{j=2}^{r} \delta\bx^{(j) \top}\bx^{(j-1)}}.
\end{align*}

    The following scenarios will be considered.
    \begin{enumerate}
        \item $r=10, d=1, \bmu_1 = (\sqrt{1/2})_{1 \times 20}, \kappa_1 = 1, \delta = 0$ (independence case);
        \item $r=10, d=1, \bmu_1 = (\sqrt{1/2})_{1 \times 20}, \kappa_1 = 1, \delta = 0.1$;
        \item $r=10, d=1, \bmu_1 = (\sqrt{1/2})_{1 \times 20}, \kappa_1 = 1, \delta = 0.5$;
        \item $r=10, d=1, \bmu_1 = (\sqrt{1/2})_{1 \times 20}, \kappa_1 = 1, \delta = 1$;
        \item $r=10, d=1, \bmu_1 = (\sqrt{1/2})_{1 \times 20}, \kappa_1 = 1, \delta = 10$.
    \end{enumerate}
\end{enumerate}

Scenario i is constructed based on the fact that conditional independence is a simple way to induce dependence between random vectors. Note that, as a particular case, when the distribution of the latent random vector $\bmu$ is degenerate (which could be interpreted as $\kappa_1 \to \infty$ or $\delta = 0$), the random vectors $\bX^{(1)}, \dots, \bX^{(r)}$ would be unconditionally iid.

Scenario ii produces a sequence of random vectors, which of them supported in a hypersphere, where the location parameter of one component $\bX^{(j)}$ given the previous ones is given by the last component $\bX^{(j-1)}$. As in the first scenario, the parameter $\delta$ represents the amount of deviation from independence.

Each scenario is simulated $100$ times for $n = 10, 50, 100$, using $B=100$ replicas in each p-value computation. Notice that scenario (a) represents the null hypothesis because, under $\delta = 0$, the components $\bX^{(1)}, \dots, \bX^{(r)}$ are independent. The adopted decision rule is to reject the null hypothesis when the p-value is less or equal to $\alpha = 0.05$. The bandwidths used in the simulation were chosen by likelihood cross-validation (see \cite{Hall1987}).

The results of the simulation are summarized in Table \ref{tab1}. As expected, the power increased as a function of $n$ - except for $\delta = 0$, which is the independence case. In such a case, the empirical power is close to 0.05. Furthermore, the rejection rate has increased as a function of $\delta$, which is reasonable, since greater values of $\delta$ represent a stronger deviation from independence.

\section{Expectation of G}

\paragraph{Asymptotic behavior of $E(\widetilde G_n^2(X_1,X_2))$.}
Recall that
\[
\widetilde H_n(x,X_1):=\int_{S^d}\bar L_h(t,x)\bar L_h(t,X_1)\,dt,
\]
and
\[
\widetilde G_n(x,y):=E\!\left[\widetilde H_n(x,X_1)\widetilde H_n(y,X_1)\right].
\]
Hence,
\[
E\!\left[\widetilde G_n^2(X_1,X_2)\right]
=
\int_{S^d}\int_{S^d}\widetilde G_n(x,y)^2 f(x)f(y)\,dx\,dy.
\]

Expanding the centered kernels, $\bar L_h=L_h-E(L_h)$, the quantity
$E(\widetilde G_n^2(X_1,X_2))$ is split into $16$ summands. We show that the first one is of
order $O(\rho(h)^{-1})$ and that the remaining terms are of smaller order. The first summand is
\[
C_1:=
\int_{S^d}\int_{S^d}
\left\{
\int_{S^d}\int_{S^d}\int_{S^d}
L_h(t,x)L_h(t,u)L_h(s,y)L_h(s,u)f(u)\,dt\,ds\,du
\right\}^2
f(x)f(y)\,dx\,dy.
\]

Define the overlap kernel
\[
H_h^*(x,u):=\int_{S^d}L_h(t,x)L_h(t,u)\,dt.
\]
Then
\[
C_1=
\int_{S^d}\int_{S^d}
\left\{
\int_{S^d}H_h^*(x,u)H_h^*(y,u)f(u)\,du
\right\}^2
f(x)f(y)\,dx\,dy.
\]

Under the null hypothesis, $f(x)=\prod_{j=1}^r f_j(x^{(j)})$, and, because $L_h$ is a product kernel,
\[
H_h^*(x,u)=\prod_{j=1}^r \varphi_{j,h_j}(x^{(j)},u^{(j)}),
\]
where
\[
\varphi_{j,h_j}(a,b):=
\int_{S^{d_j}}L_{h_j}(z,a)L_{h_j}(z,b)\,dz.
\]
Therefore,
\[
\int_{S^d}H_h^*(x,u)H_h^*(y,u)f(u)\,du
=
\prod_{j=1}^r
\left\{
\int_{S^{d_j}}
\varphi_{j,h_j}(x^{(j)},u^{(j)})
\varphi_{j,h_j}(y^{(j)},u^{(j)})
f_j(u^{(j)})\,du^{(j)}
\right\},
\]
and, consequently,
\[
C_1=\prod_{j=1}^r C_{1,j},
\]
with
\[
C_{1,j}:=
\int_{S^{d_j}}\int_{S^{d_j}}
\left\{
\int_{S^{d_j}}
\varphi_{j,h_j}(x,u)\varphi_{j,h_j}(y,u)f_j(u)\,du
\right\}^2
f_j(x)f_j(y)\,dx\,dy.
\]

Now we study $\varphi_{j,h_j}$. By definition,
\[
\varphi_{j,h_j}(a,b)
=
c_{d_j,L}(h_j)^2
\int_{S^{d_j}}
L\!\left(\frac{1-z^\top a}{h_j^2}\right)
L\!\left(\frac{1-z^\top b}{h_j^2}\right)\,dz.
\]
Applying the tangent-normal change of variables in Lemma B.4, exactly as in the derivation of the
dominant term $A_1$ in the proof of $E(\widetilde H_n^2(X_1,X_2))$, it follows that there exists a
bounded kernel $\psi_j:\mathbb R_+\to\mathbb R_+$ such that
\[
\varphi_{j,h_j}(a,b)
=
h_j^{-d_j}\psi_j\!\left(\frac{1-a^\top b}{h_j^2}\right)(1+o(1)),
\qquad h_j\to0,
\]
uniformly on $S^{d_j}\times S^{d_j}$. In particular, $\varphi_{j,h_j}$ is itself a directional kernel
with normalizing order $h_j^{-d_j}$.

Therefore, applying again the same argument used for the dominant term $A_1$ in the proof of
$E(\widetilde H_n^2(X_1,X_2))$, but now with $\varphi_{j,h_j}$ in place of $L_{h_j}$, yields
\[
C_{1,j}=O(h_j^{-d_j}).
\]
Hence,
\[
C_1
=
\prod_{j=1}^r O(h_j^{-d_j})
=
O\!\left(\prod_{j=1}^r h_j^{-d_j}\right)
=
O(\rho(h)^{-1}).
\]

Consider now any of the remaining $15$ summands in the expansion of
$E(\widetilde G_n^2(X_1,X_2))$. Each of them contains at least one factor of the form
$E\{L_h(\cdot,X)\}$ instead of a genuine overlap term. By Remark 2.1 and Lemma B.3,
\[
E\{L_h(x,X)\}=f(x)+O\!\left(\sum_{j=1}^r h_j^2\right)=O(1)
\]
uniformly in $x$, and therefore each such term loses at least one overlap-kernel contribution.
Repeating the same iterative argument as above shows that none of these remaining terms exceeds
the order of $C_1$; in fact, they are all $O(1)$ or $o(\rho(h)^{-1})$.

Thus,
\[
E\!\left[\widetilde G_n^2(X_1,X_2)\right]=O(\rho(h)^{-1}).
\]

\subsection{New proof}

\paragraph{Asymptotic behavior of $E(\widetilde G_n^2(X_1,X_2))$.}
Recall that
\[
\widetilde G_n(x,y):=E\!\left[\widetilde H_n(x,X_1)\widetilde H_n(y,X_1)\right],
\]
with
\[
\widetilde H_n(x,u):=\int_{S^d}\overline L_h(z,x)\overline L_h(z,u)\,dz,
\qquad
\overline L_h(z,x):=L_h(z,x)-E[L_h(z,X_1)].
\]
Therefore,
\[
E\!\left[\widetilde G_n^2(X_1,X_2)\right]
=
\int_{S^d}\int_{S^d}\widetilde G_n(x,y)^2 f(x)f(y)\,dx\,dy.
\]

Define
\[
m_h(x):=E[L_h(x,X_1)],
\qquad
H_n^*(x,u):=\int_{S^d}L_h(z,x)L_h(z,u)\,dz,
\]
and also
\[
B_h(x):=\int_{S^d}L_h(z,x)m_h(z)\,dz,
\qquad
C_h:=\int_{S^d}m_h(z)^2\,dz.
\]
Then
\[
\widetilde H_n(x,u)=H_n^*(x,u)-B_h(x)-B_h(u)+C_h.
\]
Hence,
\[
\widetilde G_n(x,y)
=
A_h(x,y)+R_h(x,y),
\]
where
\[
A_h(x,y):=E\!\left[H_n^*(x,X_1)H_n^*(y,X_1)\right]
=
\int_{S^d}H_n^*(x,u)H_n^*(y,u)f(u)\,du,
\]
and $R_h(x,y)$ collects all the remaining terms.

By Remark 2.1,
\[
m_h(x)=f(x)+O\!\left(\sum_{j=1}^r h_j^2\right)
\]
uniformly in $x\in S^d$. Since $L_h(\cdot,\cdot)$ is a probability kernel, it follows that
$B_h(x)=O(1)$ uniformly in $x$, and also $C_h=O(1)$. Moreover,
\[
E\!\left[H_n^*(x,X_1)\right]
=
\int_{S^d}L_h(z,x)m_h(z)\,dz
=
B_h(x)
=
O(1)
\]
uniformly in $x$. Therefore,
\[
R_h(x,y)=O(1)
\]
uniformly in $(x,y)\in S^d\times S^d$, and consequently
\[
\int_{S^d}\int_{S^d}R_h(x,y)^2 f(x)f(y)\,dx\,dy=O(1).
\]
Thus,
\[
E\!\left[\widetilde G_n^2(X_1,X_2)\right]
=
\int\!\!\int \big(A_h(x,y)+R_h(x,y)\big)^2 f(x)f(y)\,dx\,dy
\]
\[
\le
2\int\!\!\int A_h(x,y)^2 f(x)f(y)\,dx\,dy
+
2\int\!\!\int R_h(x,y)^2 f(x)f(y)\,dx\,dy
=
2A_1+O(1),
\]
where
\[
A_1:=
\int_{S^d}\int_{S^d}
\left(
\int_{S^d}H_n^*(x,u)H_n^*(y,u)f(u)\,du
\right)^2
f(x)f(y)\,dx\,dy.
\]

It remains to study $A_1$. Since $L_h$ is a product kernel, then
\[
H_n^*(x,u)
=
\prod_{j=1}^r
\varphi_{j,h_j}\!\left(x^{(j)},u^{(j)}\right),
\]
where
\[
\varphi_{j,h_j}(a,b):=
\int_{S^{d_j}}L_{h_j}(z,a)L_{h_j}(z,b)\,dz.
\]
Under the null hypothesis,
\[
f(u)=\prod_{j=1}^r f_j(u^{(j)}),
\]
and therefore
\[
A_1
=
\prod_{j=1}^r A_{1,j},
\]
with
\[
A_{1,j}:=
\int_{S^{d_j}}\int_{S^{d_j}}
\left(
\int_{S^{d_j}}
\varphi_{j,h_j}(x,u)\varphi_{j,h_j}(y,u)f_j(u)\,du
\right)^2
f_j(x)f_j(y)\,dx\,dy.
\]

Now, for each $j$, the function $\varphi_{j,h_j}$ is a kernel-overlap term of the same type as the
quantity studied in the previous subsection for $H_n^*(X_1,X_2)$. Repeating the same tangent-normal
change-of-variables argument componentwise, and using that
\[
c_{d_j,L}(h_j)\asymp h_j^{-d_j},
\]
one obtains
\[
A_{1,j}=O(h_j^{-d_j}).
\]
Hence,
\[
A_1
=
\prod_{j=1}^r O(h_j^{-d_j})
=
O\!\left(\prod_{j=1}^r h_j^{-d_j}\right)
=
O(\rho(h)^{-1}).
\]

Therefore,
\[
E\!\left[\widetilde G_n^2(X_1,X_2)\right]
\le
2A_1+O(1)
=
O(\rho(h)^{-1}),
\]
which proves the claim.

\subsection{Other new proof}

\paragraph{Asymptotic behavior of $\mathsf{E}\lrc{G_n^2(\bX_1,\bX_2)}$.}
Recall that
\[
\bar H_n(\bx,\bu) \defin \int_{\Sr} \bar L_{\bh}(\bz,\bx)\bar L_{\bh}(\bz,\bu)\,\rd \bz,
\qquad
\bar L_{\bh}(\bz,\bx) \defin L_{\bh}(\bz,\bx)-\mathsf{E}\lrc{L_{\bh}(\bz,\bX_1)},
\]
and
\[
G_n(\bx,\by) \defin \mathsf{E}\lrc{\bar H_n(\bx,\bX_1)\bar H_n(\by,\bX_1)}.
\]
Therefore,
\[
\mathsf{E}\lrc{G_n^2(\bX_1,\bX_2)}
=
\int_{\Sr}\int_{\Sr} G_n(\bx,\by)^2 f(\bx)f(\by)\,\rd \bx\,\rd \by.
\]

Define
\[
m_n(\bz) \defin \mathsf{E}\lrc{L_{\bh}(\bz,\bX_1)}
=
\int_{\Sr} L_{\bh}(\bz,\bv)f(\bv)\,\rd \bv,
\]
\[
H_n^\ast(\bx,\bu) \defin \int_{\Sr} L_{\bh}(\bz,\bx)L_{\bh}(\bz,\bu)\,\rd \bz,
\qquad
B_n(\bx) \defin \int_{\Sr} L_{\bh}(\bz,\bx)m_n(\bz)\,\rd \bz,
\qquad
C_n \defin \int_{\Sr} m_n(\bz)^2\,\rd \bz.
\]
Then
\[
\bar H_n(\bx,\bu)=H_n^\ast(\bx,\bu)-B_n(\bx)-B_n(\bu)+C_n.
\]
Hence,
\[
G_n(\bx,\by)=A_n(\bx,\by)+R_n(\bx,\by),
\]
where
\[
A_n(\bx,\by)\defin \int_{\Sr} H_n^\ast(\bx,\bu)H_n^\ast(\by,\bu)f(\bu)\,\rd \bu,
\]
and \(R_n(\bx,\by)\) collects the remaining terms.

Since \(f\) is continuous on the compact set \(\Sr\), it is bounded. Thus,
\[
0 \le m_n(\bz)=\int_{\Sr}L_{\bh}(\bz,\bv)f(\bv)\,\rd \bv
\le \norm{f}_\infty \int_{\Sr}L_{\bh}(\bz,\bv)\,\rd \bv
= \norm{f}_\infty,
\]
so \(m_n(\bz)=O(1)\) uniformly in \(\bz\in\Sr\). Consequently,
\[
B_n(\bx)=\int_{\Sr}L_{\bh}(\bz,\bx)m_n(\bz)\,\rd \bz=O(1)
\]
uniformly in \(\bx\), and also
\[
C_n=\int_{\Sr}m_n(\bz)^2\,\rd \bz=O(1).
\]
Moreover,
\[
\int_{\Sr} H_n^\ast(\bx,\bu)f(\bu)\,\rd \bu
=
\int_{\Sr}\int_{\Sr} L_{\bh}(\bz,\bx)L_{\bh}(\bz,\bu)f(\bu)\,\rd \bz\,\rd \bu
=
\int_{\Sr}L_{\bh}(\bz,\bx)m_n(\bz)\,\rd \bz
=
B_n(\bx),
\]
hence
\[
\int_{\Sr} H_n^\ast(\bx,\bu)f(\bu)\,\rd \bu=O(1)
\]
uniformly in \(\bx\). Therefore, every term entering \(R_n(\bx,\by)\) is uniformly \(O(1)\), which
implies
\[
\int_{\Sr}\int_{\Sr}R_n(\bx,\by)^2f(\bx)f(\by)\,\rd \bx\,\rd \by=O(1).
\]

By \((a+b)^2\leq 2a^2+2b^2\),
\[
\mathsf{E}\lrc{G_n^2(\bX_1,\bX_2)}
\leq
2\int_{\Sr}\int_{\Sr}A_n(\bx,\by)^2f(\bx)f(\by)\,\rd \bx\,\rd \by
+
2\int_{\Sr}\int_{\Sr}R_n(\bx,\by)^2f(\bx)f(\by)\,\rd \bx\,\rd \by.
\]
Hence,
\[
\mathsf{E}\lrc{G_n^2(\bX_1,\bX_2)} \leq 2A_{n,1}+O(1),
\]
where
\[
A_{n,1}
\defin
\int_{\Sr}\int_{\Sr}
\lrp{
\int_{\Sr}H_n^\ast(\bx,\bu)H_n^\ast(\by,\bu)f(\bu)\,\rd \bu
}^2
f(\bx)f(\by)\,\rd \bx\,\rd \by.
\]

Using again that \(f\) is bounded,
\[
\lrp{
\int_{\Sr}H_n^\ast(\bx,\bu)H_n^\ast(\by,\bu)f(\bu)\,\rd \bu
}^2
\leq
\norm{f}_\infty^2
\lrp{
\int_{\Sr}H_n^\ast(\bx,\bu)H_n^\ast(\by,\bu)\,\rd \bu
}^2.
\]
Therefore,
\[
A_{n,1}
\leq
\norm{f}_\infty^4
\int_{\Sr}\int_{\Sr}Q_n(\bx,\by)^2\,\rd \bx\,\rd \by,
\]
where
\[
Q_n(\bx,\by)\defin \int_{\Sr}H_n^\ast(\bx,\bu)H_n^\ast(\by,\bu)\,\rd \bu.
\]

Now we exploit the product structure of the kernel. Since
\[
L_{\bh}(\bx,\bu)=\prod_{j=1}^r L_{h_j}\lrp{\bx^{(j)},\bu^{(j)}},
\]
it follows that
\[
H_n^\ast(\bx,\bu)=\prod_{j=1}^r \varphi_{j,h_j}\lrp{\bx^{(j)},\bu^{(j)}},
\]
where
\[
\varphi_{j,h_j}(\ba,\bb)
\defin
\int_{\Sj}L_{h_j}(\bz,\ba)L_{h_j}(\bz,\bb)\,\rd \bz.
\]
Hence,
\[
Q_n(\bx,\by)=\prod_{j=1}^r q_{j,h_j}\lrp{\bx^{(j)},\by^{(j)}},
\]
with
\[
q_{j,h_j}(\ba,\bb)
\defin
\int_{\Sj}\varphi_{j,h_j}(\ba,\bu)\varphi_{j,h_j}(\bb,\bu)\,\rd \bu.
\]

Fix \(j\in\lrb{1,\ldots,r}\). First,
\[
\int_{\Sj}\varphi_{j,h_j}(\ba,\bu)\,\rd \bu
=
\int_{\Sj}\int_{\Sj}L_{h_j}(\bz,\ba)L_{h_j}(\bz,\bu)\,\rd \bz\,\rd \bu
=
\int_{\Sj}L_{h_j}(\bz,\ba)\,\rd \bz
=
1.
\]
Also, since \(L\) is bounded and \(c_{d_j,L}(h_j)\asymp h_j^{-d_j}\),
\[
\sup_{\ba,\bu}\varphi_{j,h_j}(\ba,\bu)
\leq
\sup_{\bz,\ba}L_{h_j}(\bz,\ba)
=
O(h_j^{-d_j}).
\]
Therefore,
\[
0\leq q_{j,h_j}(\ba,\bb)
\leq
\sup_{\bu} \varphi_{j,h_j}(\ba,\bu)\int_{\Sj}\varphi_{j,h_j}(\bb,\bu)\,\rd \bu
=
O(h_j^{-d_j}),
\]
uniformly in \((\ba,\bb)\in \Sj\times\Sj\).

Next, by Fubini's theorem and the previous normalization identity,
\[
\int_{\Sj}q_{j,h_j}(\ba,\bb)\,\rd \bb
=
\int_{\Sj}\int_{\Sj}\varphi_{j,h_j}(\ba,\bu)\varphi_{j,h_j}(\bb,\bu)\,\rd \bu\,\rd \bb
=
\int_{\Sj}\varphi_{j,h_j}(\ba,\bu)\,\rd \bu
=
1.
\]
Thus,
\[
\int_{\Sj}\int_{\Sj}q_{j,h_j}(\ba,\bb)^2\,\rd \ba\,\rd \bb
\leq
\lrp{\sup_{\ba,\bb}q_{j,h_j}(\ba,\bb)}
\int_{\Sj}\int_{\Sj}q_{j,h_j}(\ba,\bb)\,\rd \ba\,\rd \bb
=
O(h_j^{-d_j}).
\]

Using the product structure of \(Q_n\),
\[
\int_{\Sr}\int_{\Sr}Q_n(\bx,\by)^2\,\rd \bx\,\rd \by
=
\prod_{j=1}^r
\int_{\Sj}\int_{\Sj}q_{j,h_j}(\ba,\bb)^2\,\rd \ba\,\rd \bb
=
O\lrp{\prod_{j=1}^r h_j^{-d_j}}
=
O\lrp{\rho(\bh)^{-1}}.
\]
Consequently,
\[
A_{n,1}=O\lrp{\rho(\bh)^{-1}},
\]
and therefore
\[
\mathsf{E}\lrc{G_n^2(\bX_1,\bX_2)}
\leq 2A_{n,1}+O(1)
=
O\lrp{\rho(\bh)^{-1}}.
\]

This proves the required bound for \(\mathsf{E}\lrc{G_n^2(\bX_1,\bX_2)}\).

\section{Variance of $I_{n,2}$}

\begin{proof}
Recall that
\[
I_{n,2}
=
\frac{1}{n^2}\sum_{i=1}^n Z_{n,i},
\qquad
Z_{n,i}
:=
\int_{S^d} \overline L_h(\bx,\bX_i)^2 \, d\bx,
\]
where
\[
\overline L_h(\bx,\bX_i)
=
L_h(\bx,\bX_i)-E\{L_h(\bx,\bX_i)\}.
\]
Since \(Z_{n,1},\ldots,Z_{n,n}\) are iid,
\[
\Var(I_{n,2})
=
\Var\!\left(\frac{1}{n^2}\sum_{i=1}^n Z_{n,i}\right)
=
\frac{1}{n^4}\sum_{i=1}^n \Var(Z_{n,i})
=
\frac{1}{n^3}\Var(Z_{n,1})
\leq
\frac{1}{n^3}E(Z_{n,1}^2).
\]
Therefore, it is enough to prove that
\[
E(Z_{n,1}^2)=O\!\left(\rho(h)^{-2}\right).
\]

Write
\[
m_h(\bx):=E\{L_h(\bx,\bX_1)\}.
\]
Then
\[
Z_{n,1}
=
\int_{S^d}\bigl(L_h(\bx,\bX_1)-m_h(\bx)\bigr)^2\,d\bx
\leq
2\int_{S^d}L_h(\bx,\bX_1)^2\,d\bx
+
2\int_{S^d}m_h(\bx)^2\,d\bx.
\]
Hence, by \((a+b)^2\leq 2a^2+2b^2\),
\[
Z_{n,1}^2
\leq
8\left(\int_{S^d}L_h(\bx,\bX_1)^2\,d\bx\right)^2
+
8\left(\int_{S^d}m_h(\bx)^2\,d\bx\right)^2.
\]
Thus,
\[
E(Z_{n,1}^2)
\leq
8E\left[\left(\int_{S^d}L_h(\bx,\bX_1)^2\,d\bx\right)^2\right]
+
8\left(\int_{S^d}m_h(\bx)^2\,d\bx\right)^2.
\]

For the first term, note that, for any fixed \(\by\in S^d\),
\[
\int_{S^d}L_h(\bx,\by)^2\,d\bx
=
c_{d,L}(h)^2
\int_{S^d}
L\!\left(
\frac{1-\bx_1^\top\by_1}{h_1^2},
\ldots,
\frac{1-\bx_r^\top\by_r}{h_r^2}
\right)^2
\,d\bx.
\]
By the same change of variables used in the definition of \(c_{d,L}(h)\),
\[
\int_{S^d}
L\!\left(
\frac{1-\bx_1^\top\by_1}{h_1^2},
\ldots,
\frac{1-\bx_r^\top\by_r}{h_r^2}
\right)^2
\,d\bx
=
\rho(h)\lambda_{h,d}(L^2),
\]
and, since \(c_{d,L}(h)^{-1}=\rho(h)\lambda_{h,d}(L)\),
\[
\int_{S^d}L_h(\bx,\by)^2\,d\bx
=
\frac{\lambda_{h,d}(L^2)}{\rho(h)\lambda_{h,d}(L)^2}
=
O\!\left(\rho(h)^{-1}\right).
\]
Therefore,
\[
E\left[\left(\int_{S^d}L_h(\bx,\bX_1)^2\,d\bx\right)^2\right]
=
O\!\left(\rho(h)^{-2}\right).
\]

For the second term, since \(f\) is bounded on the compact set \(S^d\),
\[
m_h(\bx)
=
\int_{S^d}L_h(\bx,\by)f(\by)\,d\by
\leq
\|f\|_\infty \int_{S^d}L_h(\bx,\by)\,d\by
=
\|f\|_\infty.
\]
Hence,
\[
\int_{S^d}m_h(\bx)^2\,d\bx
\leq
\|f\|_\infty^2 \int_{S^d}d\bx
=
\omega_d\|f\|_\infty^2
=
O(1),
\]
so that
\[
\left(\int_{S^d}m_h(\bx)^2\,d\bx\right)^2
=
O(1).
\]

Combining the two bounds,
\[
E(Z_{n,1}^2)
=
O\!\left(\rho(h)^{-2}\right)+O(1)
=
O\!\left(\rho(h)^{-2}\right),
\]
because \(\rho(h)\to 0\). Consequently,
\[
\Var(I_{n,2})
\leq
\frac{1}{n^3}E(Z_{n,1}^2)
=
O\!\left(n^{-3}\rho(h)^{-2}\right).
\]
This concludes the proof.
\end{proof}
